```java
import java.util.*;
import java.io.*;

class Solution {
    public int solution(String[][] clothes) {
        int answer = 1;
        //같은 이름을 가진 의상은 존재하지 않으므로
        HashMap <String, Integer> hm = new HashMap<>();
        String clo = "";
        for(int i=0; i<clothes.length; i++){
            clo = clothes[i][1];
            if(hm.get(clo)!=null){
                hm.put(clo, hm.get(clo)+1);
            }else{
                hm.put(clo,1);
            }
        }
       
        for(String s: hm.keySet()){
            answer *= hm.get(s)+1;
        }
        //초기값 되돌리기
        answer -=1;
        
        return answer;
    }
}
```
